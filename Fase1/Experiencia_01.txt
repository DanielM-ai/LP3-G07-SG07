Experiencia_01_Sesión_01 : Números primos: Escribe un programa que incluya una función para verificar si un número es primo
y otra función para imprimir todos los números primos entre 1 y un número dado n.


import java.util.Scanner; 
public class NumerosPrimos {

	public static void main(String[] args) {
		Scanner entrada = new Scanner(System.in);
		System.out.println("Ingresa un numero: ");
		int n=entrada.nextInt();
		imprimirNumerosPrimos(n); 
	}
	
	public static boolean esPrimo(int num) {
		if(num<=1) {
			return false;
		}		//el 1er divisor probamos es 2. 
		for (int i =2; i<=num/2; i++) {	//Verificamos la existencia de divisores.
			if (num%i==0) {	// si llegas a un numero mayor que num/2 y no encontraste divisores, no tiene sentido seguir buscando. 
				return false;
			}
		}
		return true; 
	}
	
	public static void imprimirNumerosPrimos(int n) {
		for (int num=2; num<=n; num++) {
			if(esPrimo(num)) {	//llama al método esPrimo
				System.out.println(num + " es primo.");
			}
		}
	}

}
